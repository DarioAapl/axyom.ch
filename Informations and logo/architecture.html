app.js — Core Logic Pseudocode 
async function handleAsk(query) {
  const clean = sanitize(query)              // strip tags, trim
  if (clean.length < 3 || clean.length > 500) return showError("Invalid query length")

  setLoading(true)
  try {
    const res = await fetch("/api/ask", {
      method: "POST",
      headers: { "Content-Type": "application/json", "X-API-Key": getKey() },
      body: JSON.stringify({ query: clean, session_id: SESSION_ID })
    })
    if (!res.ok) throw new Error(`HTTP ${res.status}`)
    const data = await res.json()
    renderAnswer(data.answer)
    renderSources(data.sources)
  } catch (e) {
    showToast(e.message, "error")
  } finally {
    setLoading(false)
  }
}

Prompt Template — Python 
def build_prompt(query: str, chunks: list[ChunkResult]) -> list[dict]:
    context_parts = []
    for i, chunk in enumerate(chunks, 1):
        context_parts.append(
            f"SOURCE {i} [score: {chunk.score:.2f}] [url: {chunk.source_url}]\n{chunk.text}"
        )

    context_block = "\n\n".join(context_parts)
    context_block = truncate_to_tokens(context_block, max_tokens=3000)

    return [
        {
            "role": "system",
            "content": SYSTEM_PROMPT          # constant from config
        },
        {
            "role": "user",
            "content": (
                f"Context:\n{context_block}\n\n"
                f"Question: {query}"
            )
        }
    ]
⚠️ Context injection is the primary injection vector. All chunk text is sanitized at ingestion time. Query is sanitized before embedding. Both sanitization stages are independently enforced.

# ChromaDB returns cosine DISTANCE (0=identical, 2=opposite)
# Convert to similarity score [0, 1]:
def distance_to_score(distance: float) -> float:
    return 1.0 - (distance / 2.0)

# Filter below threshold
results = [r for r in results if distance_to_score(r.distance) >= SIMILARITY_THRESHOLD]

Chunking Algorithm — Pseudocode 
def recursive_chunk(text: str, chunk_size=512, overlap=64) -> list[str]:
    separators = ["\n\n", "\n", ". ", " ", ""]  # priority order
    chunks = []

    def split(text, seps):
        if token_count(text) <= chunk_size:
            return [text]
        sep = seps[0] if seps else ""
        parts = text.split(sep)
        merged = merge_splits(parts, chunk_size, overlap, sep)
        result = []
        for part in merged:
            if token_count(part) > chunk_size:
                result.extend(split(part, seps[1:]))  # recurse
            else:
                result.append(part)
        return result

    chunks = split(text, separators)
    # Deduplicate by content hash
    seen = set()
    return [c for c in chunks if (h := sha256(c)) not in seen and not seen.add(h)]

# In-memory sliding window (no Redis dep for MVP)
from collections import deque
import time

rate_store: dict[str, deque] = {}

def is_rate_limited(key: str, limit: int = 60, window_s: int = 60) -> bool:
    now = time.time()
    dq = rate_store.setdefault(key, deque())
    # Remove entries outside window
    while dq and dq[0] < now - window_s:
        dq.popleft()
    if len(dq) >= limit:
        return True
    dq.append(now)
    return False

# Drop-in pre-query hook for improved recall on short queries
class HyDEPreQueryHook:
    async def __call__(self, query: str, llm: LLMProvider) -> str:
        # Generate a hypothetical document that would answer the query
        hyp_doc = await llm.complete([
            {"role": "user", "content": f"Write a short answer to: {query}"}
        ])
        # Embed the hypothetical doc instead of the raw query
        return hyp_doc  # → passed to embed pipeline
